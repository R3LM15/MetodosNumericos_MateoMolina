def f(x):
    return x**3 - 7*x**2 + 14*x - 6  

def biseccion(a, b, tol=1e-2, max_iter=50):
    print(f"\nIntervalo inicial: [{a}, {b}]")
    if f(a) * f(b) >= 0:
        print("El método de bisección no se puede aplicar: f(a) y f(b) deben tener signos opuestos.")
        return None

    iteracion = 0
    while abs(b - a) > tol and iteracion < max_iter:
        c = (a + b) / 2.0

        if f(c) == 0:
            return c  # Se encontró la raíz exacta
        elif f(a) * f(c) < 0:
            b = c
        else:
            a = c
        iteracion += 1




        print(f"Iteración {iteracion}: c = {c}, f(c) = {f(c)} rango = [{a}, {b}]")
    return (a + b) / 2.0

        print("El método de bisección no se puede aplicar: f(a) y f(b) deben tener signos opuestos.")

    iteracion = 0
